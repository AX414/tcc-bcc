iniciar zookeeper: zookeeper-server-start.bat ..\..\config\zookeeper.properties
iniciar servidor kafka: kafka-server-start.bat ..\..\config\server.properties
cria um tópico no kafka: kafka-topics.bat --create --topic my-topic --bootstrap-server localhost:9092 --replication-factor 1 --partitions 3
permite inserir mensagens no tópico: kafka-console-producer.bat --broker-list localhost:9092 --topic my-topic
permite visualizar as mensagens que vierem no tópico: kafka-console-consumer.bat --bootstrap-server localhost:9092 --topic my-topic --from-beginning
